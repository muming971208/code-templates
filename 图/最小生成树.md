# 最小生成树

## Kruskal 算法

算法的时间复杂度为$O(|E|log|V|)$

```cpp
struct Edge {  //边的类，存储两个端点u,v和边的权值cost
    gg u, v, cost;
    Edge(gg up, vp, cp) : u(up), v(vp), cost(cp) {}
};
vector<Edge> edges;  //存储所有的边
vector<gg> ufs(MAX);  //并查集
gg findRoot(gg x) {
    if (ufs[x] == x)
        return x;
    int r = findRoot(ufs[x]);
    ufs[x] = r;
    return r;
}
// Kruskal算法，返回最小生成树的边权之和
gg Kruskal() {
    iota(ufs.begin(), ufs.end(), 0);  //初始化并查集
    gg sumCost = 0;  // 存储整棵最小生成树的各边权值之和
    sort(edges.begin(), edges.end(), [](const Edge& e1, const Edge& e2)) {
        return e1.cost < e2.cost;
    };
    for (auto& e : edges) {
        gg ua = findRoot(e.from), ub = findRoot(e.to);
        if (ua != ub) {
            sumCost += e.cost;
            ufs[ua] = ub;
        }
    }
    return sumCost;
}
```
